Basic Structure

<!-- 
Adding the controlColor="<Whatever color>" attibute is optional 
If not set, it will automatically be the default, which is light grey, unless defined otherwise in js.
-->

<div class="mSContainer">
 	<section class="mSSlide" controlColor="red">
		<!-- add text/content here -->
		<img src="img/7.jpg" class="mSSlideImage">
	<section>
	<section class="mSSlide" controlColor="#333">
		<!-- add text/content here -->
		<img src="img/8.jpg" class="mSSlideImage">
	<section>
	<section class="mSSlide" controlColor="rgb(255,212,50)">
		<!-- add text/content here -->
		<img src="img/9.jpg" class="mSSlideImage">
	<section>
</div>

<script src="js/minimalSlide.js">
Optional Settings

// js/minimalSlide.js

// all of these are settings you can change, if you would like

// stops slide show from starting automatically on load, it will start paused
ms.autoPlay = false;

// amount of time until next slide shown 
ms.delayTime = 5000;

// amount of time it takes for the new slide to fade in
ms.transitionTime = 1000;

// remove control background gradient
ms.controlGradient = false;

// sets the default color of the controls. Which will be over written
// if the controlColor attribute is set on the slide in HTML
ms.controlColorDefault = "red";
